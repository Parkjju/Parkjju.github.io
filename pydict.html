i = hash(key) % size
perturb = hash(key) # python의 perturb변수를 통해 search의 randomness 부여

while H[i] is not empty: # H[i] is occupied slot
    if H[i].key == key:
        # found!!
    # i = (i+1) % size -> linear probing
    i = (5*i+1 + perturb) % size -> 중복없이 모든 슬롯을 방문할 수 있게됨 (size가 2^n일때)
    perturb = perturb>>5
